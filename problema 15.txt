16.- Un numero natural se donmina perfecto cuando es igual a la suma de todos sus divisores , primos o no, excepto el mismo. por ejemplo; 28=1+2+4+7+14 escribr si un numero natural positivo es perfecto si o no?.

def es_perfecto(): 
    try: 
        num = int(input("Ingrese un número natural positivo: ")) 

        if num <= 0: 
            print("Por favor, ingrese un número mayor a cero.") 
            return 

        suma_divisores = 0 
         
        # Iteramos para encontrar divisores 
        for i in range(1, (num // 2) + 1): 
            if num % i == 0: 
                suma_divisores += i 
         
        if suma_divisores == num: 
            print(f"SÍ, el {num} es un número perfecto.") 
        else: 
            print(f"NO, el {num} no es un número perfecto.") 
             
    except ValueError: 
        print("Error: Debe ingresar un número entero.") 

if __name__ == "__main__": 
    es_perfecto()
