1.-Realizar un algoritmo que muestre por pantalla la tabla de multiplicar del tres y del 9.

CÓDIGO EN PYTHON: 

# Mostrar tabla del 3 
print("=== TABLA DEL 3 ===") 
for i in range(1, 13): 
    resultado = 3 * i 
    print(f"3 x {i} = {resultado}") 
print("\n=== TABLA DEL 9 ===") # \n crea un espacio entre tablas 
# Mostrar tabla del 9 
for i in range(1, 13): 
    resultado = 9 * i 
    print(f"9 x {i} = {resultado}")
